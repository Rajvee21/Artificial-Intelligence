def show(board, player):
    print(player, ":\n")
    for i in range(3):
        for j in range(9):
            if j // 3 == i:
                print(" ", board[j], end="  |")
        print()
        print("------------------")


def check_line(char, pos1, pos2, pos3):
    return pos1 == pos2 == pos3 == char


def check_all(board, char):
    if check_line(char, board[0], board[1], board[2]):
        return True
    elif check_line(char, board[3], board[4], board[5]):
        return True
    elif check_line(char, board[6], board[7], board[8]):
        return True
    elif check_line(char, board[0], board[3], board[6]):
        return True
    elif check_line(char, board[1], board[4], board[7]):
        return True
    elif check_line(char, board[2], board[5], board[8]):
        return True
    elif check_line(char, board[0], board[4], board[8]):
        return True
    elif check_line(char, board[2], board[4], board[6]):
        return True
    else:
        return False


def check_draw(board):
    
    # check if all the night cells have been used on the board (i.e., there is no number left)
    return set(board) == {"o", "x"}   


# check if the game is over.   If that is the case, return the score
def check_game_over(board):
    if check_all(board, "x"):
        return 10
    elif check_all(board, "o"):
        return -10
    elif check_draw(board):
        return 0
    else: 
        return False
    
    

def minimax(board, player):    # return the minimax score of a node
    global current
    current_score = check_game_over(board)
    
    if current_score is not False:
        return current_score
    
    # if the game is not over, do the following
    if player == "x":
        next_player = "o"
    else:
        next_player = "x"

    scores = []
    moves = []
    x_win = False
    o_win = False
    for i in range(9):   
    # check all possible moves.   
        if(board[i] == i):
            moves.append(i)
            board_copy = board.copy()
            board_copy[i] = player
            updated_score = minimax(board_copy, next_player)
            scores.append(updated_score)

            # update x_win or o_win flags
            if updated_score == 10 and player == "x":
                x_win = True
            elif updated_score == -10 and player == "o":
                o_win = True
            
            # stop searching for moves if x wins
            if x_win:
                # print("I am in x win break")
                break
            # stop searching for moves if o wins
            elif o_win:
                # print("I am in o win break")
                break
        
    if player == "x":   
    # Find the move with the highest score.   Add that move to current and return that score. 
        highest_score = max(scores)
        high_index = scores.index(highest_score)
        optimal_move = moves[high_index]
        board[optimal_move] = player
        current.append(board)
        return highest_score

    elif player == "o":  
    # Find the move with the lowest score.   Add that move to current and return that score.
        lowest_score = min(scores)
        low_index = scores.index(lowest_score)
        optimal_move = moves[low_index]
        board[optimal_move] = player
        current.append(board)
        return lowest_score



def alpha_beta_minimax(board, player, alpha=-float('inf'), beta=float('inf')):
    global current
    current_score = check_game_over(board)
    scores=[]
    
    if current_score is not False:
        return current_score
    
    if player == "x":
        next_player = "o"
        best_score = -float('inf') # initial value for alpha
    else:
        next_player = "x"
        best_score = float('inf') # initial value for beta

    moves = []
    for i in range(9):   
        if(board[i] == i):
            moves.append(i)
            board_copy = board.copy()
            board_copy[i] = player
            updated_score = alpha_beta_minimax(board_copy, next_player, alpha, beta)
            scores.append(updated_score)
            
            if player == "x":
                # update alpha value and prune if possible
                best_score = max(best_score, updated_score)
                alpha = max(alpha, best_score)
                if beta <= alpha:
                    break
            else:
                # update beta value and prune if possible
                best_score = min(best_score, updated_score)
                beta = min(beta, best_score)
                if beta <= alpha:
                    break
        
    optimal_move = moves[scores.index(best_score)]
    board[optimal_move] = player
    current.append(board)
    return best_score
